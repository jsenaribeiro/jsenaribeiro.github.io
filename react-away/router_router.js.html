/** @module Routing */
import { context } from "../shared";
import getRoute from "./getter";
import isRouted from "./routed";
import setRoute from "./setter";
/**
 * Router class
 */
export class Router {
    cursor = 0;
    routes = [];
    get routed() {
        console.log(this.routes, this.cursor);
        return this.routes[this.cursor];
    }
    getRoute(parameter) { return getRoute(parameter); }
    setRoute(route, state) {
        return setRoute(this, route, state);
    }
    /**
     * Check if the argument is routed
     * @param {string} route route to check
     */
    isRouted(route) { return isRouted(route); }
    move(type) {
        eval(`history.${type}()`);
        global.route = eval('localtion.hash').slice(1);
    }
    back() {
        if (router.cursor <= 0)
            return;
        else
            router.cursor = router.cursor - 1;
        try {
            router.move("back");
        }
        catch {
            global.route = router.routed;
        }
        finally {
            context.render();
        }
    }
    next() {
        if (router.cursor >= router.routes.length - 1)
            return;
        else
            router.cursor = router.cursor + 1;
        try {
            router.move("go");
        }
        catch {
            global.route = router.routed;
        }
        finally {
            context.render();
        }
    }
}
export const router = new Router();
