import { context } from '../shared';
/**
 * Directive for two-way data binding in form element
 * @param {object} props current props
 * @param {DirectiveArgs} args directive args
 * @returns handled bind props for two-way data binding in uncontrolled components
 */
export default function twoWayDataBindingForm(props, args) {
    if (args.tag !== "form")
        return props;
    if (!props?.bind)
        return props;
    console.log("twoWayDataBindingForm...");
    const currentOnSubmit = props.onSubmit;
    function getChildren(names, child) {
        const children = !!child?.props?.children?.map
            ? props.children.map((c) => getChildren(names, c))
            : [];
        if (names.includes(props.name))
            children.push(child);
        return children;
    }
    const onSubmit = (e) => {
        console.log("onSubmit...");
        e.preventDefault();
        e.stopPropagation();
        e.nativeEvent.submitter.onclick = "return false";
        const inputs = Array.from(e.target.elements);
        const names = inputs.map((x) => x.name);
        const childrenGetByFormNames = props.children
            .flatMap((c) => getChildren(names, c).flatMap(x => x));
        childrenGetByFormNames // validate -> setCustomValidity
            .filter((x) => x.props.validate)
            .map((x) => [x.props.name, x.props.validate()])
            .filter((array) => !!array[1])
            .forEach((array) => {
            const [field, error] = array;
            const input = inputs.find((i) => i.name == field);
            if (input)
                input.setCustomValidity(error);
        });
        const errors = inputs
            .filter((x) => !!x.validationMessage)
            .map((x) => ({
            error: x.validationMessage,
            value: x.value,
            field: x.name
        }));
        if (errors.length === 0)
            inputs.forEach(i => {
                context.enable = false;
                const field = props.bind + "." + i.name;
                context.storer(field, i.value);
                setTimeout(() => context.enable = true, 11);
            });
        return props.onCommit(errors);
    };
    return {
        ...props,
        method: "GET",
        onSubmit
    };
}
