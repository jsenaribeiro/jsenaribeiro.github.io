import { context } from '../shared';
import { ArrayProxy } from './array';
/**
 * Create proxy object to store for self-rending
 * @param store original store object
 * @param render injected component provider render
 * @returns return a self-rendering store object
 */
export function createStore(store) {
    const proxer = (obj) => new Proxy(obj, {
        set: setter,
        get: getter,
    });
    function setter(refer, field, value, proxy) {
        // console.log("setter...")
        const isArray = Array.isArray(value);
        refer[field] = isArray ? new ArrayProxy(value) : value;
        setTimeout(() => context.render(), 3);
        return true;
    }
    function getter(refer, field, proxy) {
        return refer[field];
    }
    function letter(value) {
        const proxy = {};
        for (const field in value) {
            if (Array.isArray(value[field]))
                proxy[field] = new ArrayProxy(value[field]);
            else if (typeof value[field] == "object")
                proxy[field] = letter(value[field]);
            else
                proxy[field] = value[field];
        }
        return proxer(proxy);
    }
    const proxy = letter(store);
    // console.warn(proxy)
    return proxy;
}
