/**
 * Adds a new CSS class name in current className list
 * @param props current properties
 * @param newClass new CSS class name to append
 * @returns props with new CSS class name added
 */
export function appendCss(props, newClass) {
    const noNewClass = !newClass;
    const noProps = !props;
    if (noProps || noNewClass)
        return props;
    const classes = [props.className || "", newClass || ""];
    const className = classes.join(" ");
    return { ...props, className };
}
