import { context } from "../shared";
import { dispatch } from "../storer";
/**
 * Router class
 */
export class Router {
    listeners = {};
    /**
     * Change the current route
     * @param route the new route
     * @param state state to route
     */
    getRoute(parameter) {
        const route = window.location.hash.slice(1);
        const param = {};
        const match = (x) => new RegExp(x.regex, "gi").test(route);
        const tests = context.routes.filter(match);
        const parse = (data) => {
            const content = data.toLowerCase();
            const numeric = /^\d+$/.test(content);
            const logical = /true|false/.test(content);
            const parsing = numeric || logical;
            return parsing ? JSON.parse(content) : data;
        };
        if (!parameter)
            return route;
        if (!tests || !tests.length)
            return route;
        for (const found of tests) {
            if (!found || !found.regex)
                continue;
            const regexp = new RegExp(found.regex, "gi");
            const params = [...route.matchAll(regexp)];
            if (!params)
                continue;
            for (let i = 0; i < found.names.length; i++)
                param[found.names[i].slice(1)] = parse(params[0][i + 1]);
        }
        return param[parameter] || route;
    }
    setRoute(route, state) {
        const that = this;
        const setRouteString = (route) => {
            const listeners = context.router.listeners;
            if (listeners[route]?.length > 0)
                listeners[route].forEach(f => f());
            window.location.hash = route;
            dispatch(route, state);
            context.render();
        };
        const setRouteNumber = (index) => {
            if (index > 0)
                while (index-- > 0)
                    window.history.go();
            if (index < 0)
                while (index++ < 0)
                    window.history.back();
            context.render();
        };
        typeof route == "string" ? setRouteString(route)
            : typeof route == "number" ? setRouteNumber(route)
                : context.render();
    }
    /**
     * Check if the argument is routed
     * @param route route to check
     */
    isRouted(route) {
        if (!route)
            return false;
        const current = window.location.hash.slice(1);
        const routing = current.split('/');
        const queried = route.split('/');
        if (current === "/")
            return route === current;
        for (let i = 0; i < queried.length; i++) {
            if (routing[i] == undefined)
                continue;
            if (queried[i] == undefined)
                continue;
            if (routing[i].length == 0)
                continue;
            if (queried[i][0] === ":")
                continue;
            if (routing[i] !== queried[i])
                return false;
        }
        return true;
    }
}
