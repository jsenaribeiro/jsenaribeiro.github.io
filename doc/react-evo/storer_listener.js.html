import { context } from "../shared";
const events = {};
const eventLabels = ["mounted", "updated", "unmount", "catched", "refresh"];
/**
 * Multi-event dispatcher
 * @param evt event label
 * @param arg event value
 * @returns false if the event is not found
 */
export function dispatch(evt, arg = {}) {
    if (!events[evt])
        return false;
    else
        events[evt].forEach(f => f(arg));
}
/**
 * Multi-event listener
 * @param evt event label
 * @param fnc event handler
 * @returns event canceller
 */
export function listener(evt, fnc) {
    if (!events[evt])
        events[evt] = [];
    events[evt].push(fnc);
    return () => events[evt]
        .filter(f => f.toString() === fnc.toString())
        .forEach((_, i) => delete events[i]);
}
/**
 * Event registry
 * @param add is add (else remove)?
 */
export function register(add) {
    const listening = add
        ? window.addEventListener
        : window.removeEventListener;
    const registration = (lbl, fns) => fns.forEach(f => listening(lbl, f));
    const isDOMevent = (event) => typeof document.body["on" + event] !== undefined;
    Object.keys(events)
        .filter(k => isDOMevent(k))
        .filter(k => !k.startsWith("/"))
        .filter(k => !eventLabels.includes(k))
        .forEach(k => registration(k, events[k]));
    if (add)
        window.addEventListener("hashchange", x => render());
    else
        window.removeEventListener("hashchange", x => render());
    window.onbeforeunload = function (e) {
        console.log("onbeforeunload");
    };
    const render = () => {
        context.render();
    };
}
