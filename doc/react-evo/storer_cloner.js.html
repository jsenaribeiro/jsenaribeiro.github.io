import React from "react";
import { inRouted } from '../router/routed';
import { focus } from './focus';
import { DIRECTIVES } from "../evo";
import { createProxy } from "./proxy";
var index = 0;
/**
 * Deep component cloning
 * @param {object} root root component
 * @param {object} tag initial tag name
 * @returns deep cloned component three
 */
export const cloneComponent = (root, tag = "App") => (index = 0) || focus(() => cloneChildren(root.props.children, tag, "0"));
const clone = (child, tag, uid) => child.props?.shown === false ? undefined
    : React.cloneElement(deepClone(child, tag, uid), {}, cloneChildren(child.props.children, tag, uid));
const cloneChild = (child, tag, key) => !inRouted(child) ? undefined
    : React.isValidElement(child)
        ? clone(child, tag, key)
        : child;
const cloneChildren = (children, tag, key) => React.Children.map(children, (child) => cloneChild(child, tag, key));
const deepClone = (child, tag, key) => !DIRECTIVES?.length ? child
    : when(child, "class") ? child
        : when(child, "element") ? cloneElement(child, tag, key)
            : when(child, "fragment") ? cloneElement(child, tag, key)
                : when(child, "function") ? { ...child, type: (_) => cloneFunction(child, tag) }
                    : child;
function cloneFunction(child, parent) {
    const identity = `${child.type.name}-${index}`;
    const tagLabel = child.type.name;
    const oldProps = child.props;
    const argStore = createProxy(window.store);
    const newChild = child.type(oldProps);
    const newProps = properHandler(newChild.props, tagLabel, parent, identity);
    // console.log(child.props.route)
    // child.props.route && dispatch(child.props.route)
    const children = React
        .Children.map(newProps.children, (c) => deepClone(c, tagLabel, identity));
    return ({ ...newChild, props: { ...newProps, children } });
}
function cloneElement(child, tag, key) {
    const newProps = properHandler(child.props, child.type, tag, key);
    const children = cloneChildren(newProps.children, tag, key);
    return { ...child, props: { ...newProps, children } };
}
const when = (child, ...types) => types.includes("class") && child?.type?.toString().startsWith("class")
    || types.includes("fragment") && child?.type?.toString() == 'Symbol(react.fragment)'
    || types.includes("element") && typeof child?.type === "string"
    || types.includes((typeof child?.type));
export const properHandler = (props, tag, componentTag, key) => DIRECTIVES
    .map(fn => fn(props, { tag, componentTag, uid: key }))
    .reduce((last, next) => ({ ...last, ...next }));
