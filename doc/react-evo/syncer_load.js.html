/** @module Synchronizer */
import { syncher } from ".";
import { context } from "../shared";
import { error } from "./functions";
/**
 * Loading request by HTTP get verb
 * @param {ApiAsync} that ApiSync context
 * @param {object} filter query object for query string GET request
 * @returns responde queried from RESTful API
 */
export default async function loadAsync(that, filter) {
    const headers = await that.headers();
    const queries = filter && "?" + new URLSearchParams(filter);
    const settings = { headers, signal: syncher.canceller.signal };
    const route = that.config.search?.route || that.config.route;
    const url = that.config.url + route + queries;
    syncher.isLoading = true;
    context.render();
    try {
        const response = await fetch(url, settings);
        const result = await response.json();
        that.config.set(window.store, result);
        return result;
    }
    catch (ex) {
        console.log(0, ex);
        syncher.exception = that.config.onError(error);
    }
    finally {
        syncher.outOfSync = false;
        syncher.isLoading = false;
        context.render();
    }
}
