import React from "react";
import { cloneComponent } from './cloner';
import { createProxy } from './proxy';
import { context } from "../shared";
import { dispatch, register } from './listener';
import { getRoute } from "../router";
/** Multi-functional ReactEvo provider */
export class Provider extends React.Component {
    constructor(props) {
        super(props);
        const refresh = () => this.setState({ tick: new Date().getTime() });
        window.store = createProxy(props.src);
        context.render = (timeout = 1) => !context.enable ? undefined
            : setTimeout(() => refresh(), timeout);
        dispatch(getRoute());
    }
    render() {
        dispatch("refresh");
        // console.log("render...")
        return cloneComponent(this);
    }
    componentDidMount() { register(true); dispatch("mounted"); }
    componentWillUnmount() { register(false); dispatch("unmount"); }
    componentDidCatch = (error, info) => dispatch("catched", { error, errorInfo: info });
    componentDidUpdate = (props, state, snapshot) => dispatch("updated", { prevProps: props, prevState: state, snapshot });
}
