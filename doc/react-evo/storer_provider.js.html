import React from "react";
import { cloneComponent } from './cloner';
import { context } from "../shared";
import { createStore } from './store';
/**
 * Global self-rendering store object.
 * This object is the core object of ReactEvo state management system.
 * All the domain data is globally handled by this object, and
 * any time it is changed, the host component will starts the top-down rendering
 * by root compnent (instead the current component, as in standard React way).
 */
export var store = {};
/** Multi-functional ReactEvo provider */
export class Provider extends React.Component {
    constructor(props) {
        super(props);
        const refresh = () => this.setState({ tick: new Date().getTime() });
        store = createStore(props.src);
        context.render = (timeout = 1) => !context.enable ? undefined
            : setTimeout(() => refresh(), timeout);
    }
    render() { return cloneComponent(this); }
}
