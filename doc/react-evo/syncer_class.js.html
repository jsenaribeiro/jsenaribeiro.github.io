/** @namespace Synchronizer */
import { defineSync, error } from './functions';
import { login } from '../auther';
import { parseJWT } from '../auther/login';
import saveAsync from './save';
import loadAsync from './load';
var that = {};
/**
 * Synchronizer configurator class
 */
export class ApiAsync {
    reload = false;
    config;
    synced = undefined;
    actual = undefined;
    isEqual = (x, y) => JSON.stringify(x) === JSON.stringify(y);
    hasIdField = (x) => !this.config.avoidId && Object.keys(x).includes(this.config.idField);
    constructor(settings) {
        if (!settings.url)
            throw error("URL is required");
        if (!settings.get)
            throw error("Get is required");
        if (!settings.set)
            throw error("Set is required");
        that = this;
        this.config = {
            ...defineSync(settings),
            search: settings.search && defineSync(settings.search),
            create: settings.create && defineSync(settings.create),
            update: settings.update && defineSync(settings.update),
            delete: settings.delete && defineSync(settings.delete)
        };
        fetch(settings.url).then(x => x.json()).then(x => this.synced = x);
    }
    /** Synchronization builder */
    build = () => ({
        saveAsync: (reload) => saveAsync(that, reload),
        loadAsync: (filter) => loadAsync(that, filter || {}),
        onSave: (e) => saveAsync(that, false),
        onLoad: (e) => loadAsync(that, {})
    });
    async headers() {
        const headers = new Headers();
        const jsonToken = sessionStorage.getItem("token") || "";
        const token = jsonToken && parseJWT(jsonToken);
        const expired = token ? token.exp * 1000 < Date.now() : false;
        if (expired) {
            try {
                await login();
            }
            catch {
                sessionStorage.removeItem("token");
            }
        }
        headers.append("Authorization", jsonToken);
        headers.append("Content-Type", "application/json; charset=UTF-8");
        return headers;
    }
}
