import React from "react";
import { inRouted } from '../router/routed';
import { commonDirectives } from './directives';
import { customDirectives } from "../storer/startup";
import { refocus, validateDirective } from "./helpers";
/**
 * @name cloneComponent
 * @param props current react component props
 * @param reprops handle the children props before render
 * @param unrouted checks if a routed component is unrouted
 * @returns cloned children
 */
export const cloneComponent = (root) => refocus(() => cloneChildren(root.props.children));
const clone = (child, tag) => React.cloneElement(deepClone(child, tag), {}, cloneChildren(child.props.children, tag));
const cloneChild = (child, tag = "App") => !inRouted(child) ? undefined
    : React.isValidElement(child)
        ? clone(child, tag)
        : child;
const cloneChildren = (children, tag = "App") => React.Children.map(children, (child) => cloneChild(child, tag));
const deepClone = (child, tag) => when(child, "class") ? child
    : when(child, "element") ? cloneElement(child, tag)
        : when(child, "fragment") ? cloneElement(child, tag)
            : when(child, "function") ? { ...child, type: (_) => cloneFunction(child) }
                : child;
function cloneFunction(child) {
    const tagLabel = child.type.name;
    const oldProps = child.props;
    const oldChild = child.type(oldProps);
    const newChild = changer(oldChild, tagLabel);
    const newProps = newChild.props;
    const children = React
        .Children.map(newProps.children, (c) => deepClone(c, tagLabel));
    return ({ ...newChild, props: { ...newProps, children } });
}
function cloneElement(child, tag) {
    const newChild = changer(child, tag);
    const children = cloneChildren(newChild.props.children, tag);
    return { ...newChild, props: { ...newChild.props, children } };
}
const changer = (child, tag) => commonDirectives
    .concat(customDirectives)
    .filter(validateDirective)
    .map(f => child = f(child, tag))
    .reduce(_ => child);
const when = (child, ...types) => types.includes("class") && child.type?.toString().startsWith("class")
    || types.includes("fragment") && child.type.toString() == 'Symbol(react.fragment)'
    || types.includes("element") && typeof child.type === "string"
    || types.includes((typeof child.type));
