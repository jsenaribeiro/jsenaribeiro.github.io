/** @module synching */
import { syncher } from '.';
import { context, delay } from '../shared';
import { error } from './functions';
import loadAsync from './load';
const FAILED_FETCH_SYNC = "Sync failed: not possible to fetch data...";
const same = (x, y) => JSON.stringify(x) === JSON.stringify(y);
const diff = (x, y) => same(x, y) === false;
async function preValidationAsync(that, retries = 3) {
    const ready = !!that.synced;
    const failed = !that.synced && !retries;
    const retry = () => preValidationAsync(that, retries - 1);
    if (failed)
        throw FAILED_FETCH_SYNC;
    if (!ready)
        await delay(500, async () => await retry());
}
/**
 * Change request inference for POST, PUT and DELETE verbs
 * @param {ApiAsync} that ApiAsync context
 * @param {boolean} reload send a GET request after a change request?
 */
export default async function saveAsync(that, reload) {
    await preValidationAsync(that);
    const value = that.config.get(window.store);
    const state = value?.array ? value.array : value;
    that.reload = reload;
    that.actual = JSON.parse(JSON.stringify(state));
    const guid = that.config.idField;
    const unit = !Array.isArray(that.actual);
    if (unit && !that.actual[guid])
        action(that, "create", that.actual);
    else if (unit && that.actual[guid]) {
        const exists = unit
            ? JSON.stringify(that.synced) === JSON.stringify(that.actual)
            : that.synced.some((x) => x[guid] === that.actual[guid]);
        if (exists)
            action(that, "update", that.actual);
    }
    if (!unit && that.actual) {
        const createds = [];
        const deleteds = [];
        const updateds = [];
        that.actual.forEach((item) => {
            const creating = that.synced
                .every((x) => x[guid] !== item[guid]);
            const updating = that.synced
                .filter((x) => x[guid] === item[guid])
                .some((x) => diff(x, item));
            if (creating)
                createds.push(item);
            if (updating)
                updateds.push(item);
        });
        that.synced.forEach((item) => {
            const identity = (x) => x[guid] !== item[guid];
            const deleting = item[guid] && that.actual.every(identity);
            if (deleting)
                deleteds.push(item);
        });
        createds.forEach(x => action(that, "create", x));
        updateds.forEach(x => action(that, "update", x));
        deleteds.forEach(x => action(that, "delete", x));
        const noAction = createds.length
            + updateds.length
            + deleteds.length === 0;
        if (noAction && that.reload)
            loadAsync(that, {});
    }
}
/**
 * Executes the change request for POST, PUT and DELETE verbs
 * @param {ApiAsync} that ApiAsync context
 * @param {Action} action change request label (create, update, delete)
 * @param {object} entity target entity object
 */
async function action(that, action, entity) {
    const id = entity[that.config.update?.idField || that.config.idField];
    const method = action == "create" ? "POST"
        : action == "update" ? "PUT"
            : "DELETE";
    const hasId = action === "create" ? false
        : that.config[action] ? that.config[action].inIdURL
            : that.config.inIdURL;
    const headers = await that.headers();
    const route = that.config[action]?.route || that.config.route;
    const url = that.config.url + route + (hasId ? `/${id}` : "").replace("//", "/");
    const body = JSON.stringify(entity);
    const settings = {
        method, body, headers,
        signal: syncher.canceller.signal
    };
    syncher.isLoading = true;
    context.render();
    try {
        const response = await fetch(url, settings);
        if (response.ok)
            await response.json();
        else
            throw { message: response.statusText };
        that.synced = that.actual;
        if (that.reload)
            loadAsync(that, {});
    }
    catch (ex) {
        syncher.exception = that.config.onError(error) || ex.message;
    }
    finally {
        syncher.outOfSync = false;
        syncher.isLoading = false;
        context.render();
    }
}
