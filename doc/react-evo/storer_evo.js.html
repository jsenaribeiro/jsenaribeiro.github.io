import { jsx as _jsx } from "react/jsx-runtime";
import React from 'react';
import ReactDOM from 'react-dom/client';
import { Provider } from './provider';
import { I18nEVO } from './i18n';
import formerProper from '../binder';
import stylerProper from '../styler';
export const DIRECTIVES = [
    formerProper,
    stylerProper,
];
/**
 * React EVO startup class
 */
export default class ReactEVO {
    static createRoot(strict, query, app, context, directives = []) {
        if (!!directives.length)
            directives.forEach((d) => DIRECTIVES.push(d));
        const root = document.querySelector(query);
        const wrap = _jsx(Provider, { src: context, children: app });
        const main = strict ? _jsx(React.StrictMode, { children: wrap }) : wrap;
        ReactDOM.createRoot(root).render(main);
        return new I18nEVO();
    }
}
