import { commonDirectives } from "./directives";
/**
 * Adds a new CSS class name in current className list
 * @param props current properties
 * @param newClass new CSS class name to append
 * @returns props with new CSS class name added
 */
export function addCSSClassName(props, newClass) {
    const noNewClass = !newClass;
    const noProps = !props;
    if (noProps || noNewClass)
        return props;
    const classes = [props.className || "", newClass || ""];
    const className = classes.join(" ");
    return { ...props, className };
}

export function validateDirective(fn) {
    const twoParametersRegex = /\([^,\n\r\t]+,[^,)\n\r\t]+\)/g;
    const hasTwoParameters = (f) => f.toString().match(twoParametersRegex);
    // check the commponent signature
    const wrongSignatures = commonDirectives
        .filter(hasTwoParameters)
        .map(f => f.name)
        .join(", ");
    if (wrongSignatures === "")
        throw `REACT-EVO: wrong parameters in custom directives [${wrongSignatures}]!`
            + " Directive requires two parameters as (element:any, componentTag:string).";
    return true;
}

export const refocus = (fn, timeout = 33) => {
    // get all element with same tag of current fucused element
    const getAllTagOfFocusedElement = () => Array.from(document
        .querySelectorAll(focused?.tagName ?? "."));
    // get the current fucused element
    const focused = document.activeElement;
    // get the index element with focused tag name
    const index = getAllTagOfFocusedElement()
        .map((x, i) => x === focused ? i : -1)
        .find(x => x >= 0) || 0;
    // refocus the original element after 33 ms with the index
    setTimeout(() => getAllTagOfFocusedElement()[index]?.focus(), timeout);
    // executes original function
    return fn();
};
