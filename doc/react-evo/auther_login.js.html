/** @module Authentication */
import { setRoute } from '../router';
import { content } from '../shared';
const INVALID = "invalid";
const UNAUTHORIZED = "unauthorized";
/**
 * Get the current decoded token
 * @returns decoded token
 */
export function logon() {
    const token = sessionStorage.getItem("token");
    return token ? parseJWT(token) : undefined;
}
/**
 * Remove current logged user token
 */
export function logout() {
    sessionStorage.removeItem("token");
    setRoute("/");
    // context.render()
}
export async function login(username, password, timeout = 1000, retry = 7) {
    const none = username === undefined && password === username;
    const credential = none
        ? sessionStorage.getItem("login") || ""
        : content.setting.encript
            ? btoa(`${username}:${password}`)
            : JSON.stringify({ username, password });
    if (!credential)
        throw new Error("login: no credential");
    const headers = new Headers();
    headers.append("content-type", "application/json; charset=UTF-8");
    if (content.setting.method === "GET")
        headers.append("Authorization", `Basic ${credential}`);
    const body = content.setting.method === "POST" ? credential : undefined;
    try {
        const settings = { body, headers };
        const response = await fetch(content.setting.address, settings);
        if (response.status === 401)
            throw UNAUTHORIZED;
        if (response.ok === false)
            throw INVALID;
        const value = await response.json();
        const token = content.setting.getToken(value);
        sessionStorage.setItem("token", token);
        sessionStorage.setItem("login", credential);
        setRoute(content.setting.routes.success);
    }
    catch (ex) {
        console.error("login", ex);
        const relogin = () => username && password
            ? login(username, password, timeout * 2)
            : login();
        if (ex === UNAUTHORIZED && content.setting.relogin)
            setTimeout(relogin, timeout);
        else
            setRoute(content.setting.routes.failure);
    }
    finally {
        return content.setting;
    }
}
export function parseJWT(token) {
    if (!token)
        return {};
    const parseJwtArray = (c) => '%' + ('00' + c.charCodeAt(0).toString(16)).slice(-2);
    const address = token.split('.')[1];
    const base64 = address.replace(/-/g, '+').replace(/_/g, '/');
    const array = window.atob(base64).split('');
    const payload = decodeURIComponent(array.map(parseJwtArray).join(''));
    return JSON.parse(payload);
}
